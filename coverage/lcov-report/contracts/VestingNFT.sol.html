// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
 
import "../node_modules/@openzeppelin/contracts/token/ERC721/IERC721.sol";
import "../node_modules/@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "../node_modules/@openzeppelin/contracts/utils/math/SafeMath.sol";
 
contract NFTVesting {
    using SafeMath for uint256;
 
    IERC721 public nftContract;
    IERC20 public vestingTokenContract;
 
    uint256 public constant VESTING_PERIOD = 30 days;
 
    struct VestingInfo {
        uint256 startTime;
        uint256 amount;
        uint256 claimedAmount;
    }
 
    mapping(uint256 => VestingInfo) public vestingInfo;
    mapping(uint256 => mapping(address => uint256)) public fractionalizedShares;
    mapping(uint256 => mapping(address => uint256))
        public fractionalizedVestingAmount;
    mapping(uint256 => mapping(address => uint256))
        public fractionalizedClaimedAmount;
 
    event Deposit(uint256 indexed tokenId, uint256 amount);
    event Claim(uint256 indexed tokenId, uint256 amount);
 
    constructor(address _nftContract, address _vestingTokenContract) {
        nftContract = IERC721(_nftContract);
        vestingTokenContract = IERC20(_vestingTokenContract);
    }
 
    function deposit(uint256 tokenId, uint256 amount) external {
        require(nftContract.ownerOf(tokenId) != address(0));
 
        vestingTokenContract.transferFrom(msg.sender, address(this), amount);
        vestingInfo[tokenId] = VestingInfo(block.timestamp, amount, 0);
        
        emit Deposit(tokenId, amount);
    }
 
    function claim(uint256 tokenId) external {
        require(nftContract.ownerOf(tokenId) == msg.sender, "Not the owner");
 
        VestingInfo storage info = vestingInfo[tokenId];
        require(info.startTime != 0, "No vesting info");
 
        uint256 elapsedTime = block.timestamp.sub(info.startTime);
        uint256 vestedAmount = (VESTING_PERIOD < elapsedTime)
            ? info.amount
            : elapsedTime.mul(info.amount).div(VESTING_PERIOD);
        uint256 claimableAmount = vestedAmount.sub(info.claimedAmount);
        require(claimableAmount > 0, "Nothing to claim");
 
        info.claimedAmount = vestedAmount;
        vestingTokenContract.transfer(msg.sender, claimableAmount);
 
        emit Claim(tokenId, claimableAmount);
    }
 
    function fractionalizedClaim(uint256 tokenId) external {
        VestingInfo storage info = vestingInfo[tokenId];
        require(info.startTime != 0, "No vesting info");
 
        uint256 share = fractionalizedShares[tokenId][msg.sender];
        require(share > 0, "No fractionalized share");
 
        uint256 elapsedTime = block.timestamp.sub(info.startTime);
        uint256 vestedAmount = (VESTING_PERIOD < elapsedTime)
            ? fractionalizedVestingAmount[tokenId][msg.sender]
            : elapsedTime
                .mul(fractionalizedVestingAmount[tokenId][msg.sender])
                .div(VESTING_PERIOD);
        uint256 fractionalizedClaimableAmount = vestedAmount.sub(
            fractionalizedClaimedAmount[tokenId][msg.sender]
        );
        require(fractionalizedClaimableAmount > 0, "Nothing to claim");
 
        info.claimedAmount = info.claimedAmount.add(
            fractionalizedClaimableAmount
        );
        vestingTokenContract.transfer(
            msg.sender,
            fractionalizedClaimableAmount
        );
 
        fractionalizedClaimedAmount[tokenId][msg.sender] = vestedAmount;
        emit Claim(tokenId, fractionalizedClaimableAmount);
    }
 
    function onFractionalization(
        uint256 tokenId,
        address[] calldata holders,
        uint256[] calldata shares
    ) external {
        VestingInfo storage info = vestingInfo[tokenId];
        // Ensure the holders and shares arrays have the same length
        require(
            holders.length == shares.length,
            "Holders and shares arrays must have the same length"
        );
 
        // Ensure the tokenId exists in the vestingInfo mapping
        require(
            vestingInfo[tokenId].startTime != 0,
            "TokenId not found in vestingInfo"
        );
        uint256 vestingAmount = (info.amount).sub(info.claimedAmount);
        // Clear any existing fractionalizedShares for the tokenId
        for (uint256 i = 0; i < holders.length; i++) {
            fractionalizedShares[tokenId][holders[i]] = 0;
            fractionalizedClaimedAmount[tokenId][holders[i]] = 0;
            fractionalizedVestingAmount[tokenId][holders[i]] = 0;
        }
 
        // Update the fractionalizedShares mapping with the new shares
        for (uint256 i = 0; i < holders.length; i++) {
            fractionalizedShares[tokenId][holders[i]] = shares[i];
            fractionalizedVestingAmount[tokenId][holders[i]] = vestingAmount
                .mul(shares[i])
                .div(100);
        }
    }
}
