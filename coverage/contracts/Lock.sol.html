// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
 
import "../node_modules/@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "../node_modules/@openzeppelin/contracts/token/ERC20/IERC20.sol";
 
contract VestingNFT is ERC721 {
    IERC20 public rentToken;
    uint256 public constant VESTING_PERIOD = 30 days;
 
    struct Vesting {
        uint256 amount;
        uint256 startTime;
        uint256 claimedAmount;
    }
 
    mapping(uint256 => Vesting) public vestings;
    event VestingDeposited(address indexed owner, uint256 indexed tokenId, uint256 amount);
//    event VestingClaimed(address indexed owner, uint256 indexed tokenId, uint256 amount);
//    event FractionalVestingClaimed(address indexed owner, uint256 indexed tokenId, uint256 amount);
 
    constructor(address rentTokenAddress) ERC721("VestingNFT", "VNFT") {
        rentToken = IERC20(rentTokenAddress);
    }
    function createToken(address recipient, uint256 tokenId) external {
    // Check that the token does not already exist
    require(!_exists(tokenId), "Token already exists");
 
    // Mint the token using safeMint
    _safeMint(recipient, tokenId);
}
    function depositRent(uint256 tokenId, uint256 amount) external {
        require(_exists(tokenId), "Token does not exist");
       // console.log(rentToken.balanceOf(msg.sender),rentToken.balanceOf(address(this)));
        rentToken.transferFrom(msg.sender, address(this), amount);
       // console.log(rentToken.balanceOf(msg.sender),rentToken.balanceOf(address(this)));
        Vesting storage vesting = vestings[tokenId];
        vesting.amount = amount;
        vesting.startTime = block.timestamp;
        vesting.claimedAmount = 0;
        emit VestingDeposited(msg.sender, tokenId, amount);
        
    }
 
    function claimVested(uint256 tokenId) external {
        require(_exists(tokenId), "Token does not exist");
        require(ownerOf(tokenId) == msg.sender, "Not the owner");
 
        Vesting storage vesting = vestings[tokenId];
        uint256 elapsedTime = block.timestamp - vesting.startTime;
        uint256 vestedAmount = (elapsedTime * vesting.amount) / VESTING_PERIOD;
        uint256 claimableAmount = vestedAmount - vesting.claimedAmount;
 
        require(claimableAmount > 0, "Nothing to claim");
 
        vesting.claimedAmount = vestedAmount;
        rentToken.transfer(msg.sender, claimableAmount);
    }
 
    function onFractionalization(uint256 tokenId, uint256[] memory newTokenIds, uint256[] memory shares) external {
        require(_exists(tokenId), "Token does not exist");
        require(newTokenIds.length == shares.length, "Invalid input");
 
        Vesting storage originalVesting = vestings[tokenId];
        uint256 totalShares = 0;
        for (uint256 i = 0; i < shares.length; i++) {
            totalShares += shares[i];
        }
 
        for (uint256 i = 0; i < newTokenIds.length; i++) {
            uint256 newTokenId = newTokenIds[i];
            uint256 share = shares[i];
 
            Vesting storage newVesting = vestings[newTokenId];
            newVesting.amount = (originalVesting.amount * share) / totalShares;
            newVesting.startTime = originalVesting.startTime;
            newVesting.claimedAmount = (originalVesting.claimedAmount * share) / totalShares;
        }
    }
}
